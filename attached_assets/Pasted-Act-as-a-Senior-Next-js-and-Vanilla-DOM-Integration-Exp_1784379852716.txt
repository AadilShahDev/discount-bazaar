TypeScript
"use client";
import React, { useEffect, useRef } from 'react';
import '@sfpy/atoms'; // Registers the web components globally

export default function SafepayCardAtom({ 
  tracker, 
  authToken, 
  imperativeRef, 
  inputStyle,
  onReady,
  onValidated,
  onError,
  onProceedToAuthentication // if handling 3DS
}: any) {
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    // 1. Guard clause: Ensure we have the required data and a mounted container
    if (!tracker || !authToken || !containerRef.current) return;

    // 2. Clear any existing children to prevent React StrictMode duplication
    containerRef.current.innerHTML = '';

    // 3. Create the raw Web Component for Card Capture
    const cardAtom = document.createElement('safepay-card-atom') as any;

    // 4. Assign configuration directly to the DOM properties
    cardAtom.environment = 'sandbox';
    cardAtom.tracker = tracker;
    cardAtom.authToken = authToken;
    if (inputStyle) cardAtom.inputStyle = inputStyle;

    // 5. Attach callbacks directly as properties (per Safepay NPM documentation)
    if (onReady) cardAtom.onReady = onReady;
    if (onValidated) cardAtom.onValidated = (e: any) => onValidated(e);
    if (onError) cardAtom.onError = (e: any) => onError(e);

    // 6. Connect the imperative ref so the parent can trigger submissions
    if (imperativeRef) {
      imperativeRef.current = {
        submit: () => cardAtom.submit(),
        validate: () => cardAtom.validate(),
        fetchValidity: () => cardAtom.fetchValidity(),
      };
    }

    // 7. Create the Payer Authentication Atom (for 3D-Secure/OTP flows)
    const authAtom = document.createElement('safepay-payer-auth-atom') as any;
    authAtom.environment = 'sandbox';
    authAtom.tracker = tracker;
    authAtom.authToken = authToken;
    if (onProceedToAuthentication) authAtom.onProceedToAuthentication = onProceedToAuthentication;

    // 8. Inject the components securely into the DOM
    containerRef.current.appendChild(cardAtom);
    containerRef.current.appendChild(authAtom);

    // 9. Cleanup function
    return () => {
      if (containerRef.current) containerRef.current.innerHTML = '';
    };
  }, [tracker, authToken, imperativeRef, inputStyle, onReady, onValidated, onError, onProceedToAuthentication]);

  return <div ref={containerRef} className="w-full min-h-[150px] relative" />;
}
