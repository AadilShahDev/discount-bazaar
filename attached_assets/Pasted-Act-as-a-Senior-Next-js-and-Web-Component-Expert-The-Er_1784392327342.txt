"use client";
import React, { useEffect, useRef, useState } from 'react';
import Script from 'next/script';

export default function SafepayCardAtom({ 
  tracker, 
  authToken, 
  imperativeRef, 
  onReady, 
  onValidated, 
  onError, 
  onProceedToAuthentication 
}: any) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [scriptLoaded, setScriptLoaded] = useState(false);

  useEffect(() => {
    // Only inject when the CDN script is loaded and tokens are ready
    if (!scriptLoaded || !tracker || !authToken || !containerRef.current) return;

    const container = containerRef.current;
    container.innerHTML = ''; // Clear strict-mode duplicates

    // 1. Create the raw Web Component directly
    const cardAtom = document.createElement('safepay-card-atom') as any;
    
    // 2. Assign JS properties (camelCase, as per Safepay docs)
    cardAtom.environment = 'sandbox';
    cardAtom.tracker = tracker;
    cardAtom.authToken = authToken;
    cardAtom.inputStyle = { 
      fontFamily: 'Inter, system-ui, sans-serif', 
      fontSize: '16px', 
      color: '#111827' 
    };

    // 3. Attach callbacks as properties
    cardAtom.onReady = onReady;
    cardAtom.onValidated = (e: any) => onValidated?.(e);
    cardAtom.onError = (e: any) => onError?.(e);

    // 4. Connect imperative methods
    if (imperativeRef) {
      imperativeRef.current = {
        submit: () => cardAtom.submit(),
        validate: () => cardAtom.validate(),
        fetchValidity: () => cardAtom.fetchValidity(),
      };
    }

    // 5. Create Payer Authentication Atom
    const authAtom = document.createElement('safepay-payer-auth-atom') as any;
    authAtom.environment = 'sandbox';
    authAtom.tracker = tracker;
    authAtom.authToken = authToken;
    if (onProceedToAuthentication) {
      authAtom.onProceedToAuthentication = onProceedToAuthentication;
    }

    // 6. Inject natively into the DOM
    container.appendChild(cardAtom);
    container.appendChild(authAtom);

    return () => {
      container.innerHTML = '';
    };
  }, [scriptLoaded, tracker, authToken]); // Intentionally omitting callbacks to prevent re-renders

  return (
    <>
      <Script 
        src="https://cdn.jsdelivr.net/npm/@sfpy/atoms@latest/dist/components/index.global.js" 
        strategy="afterInteractive"
        onLoad={() => setScriptLoaded(true)}
      />
      <div ref={containerRef} className="w-full min-h-[150px] relative">
        {!scriptLoaded && (
          <div className="w-full h-full min-h-[150px] flex items-center justify-center text-sm text-gray-500 animate-pulse">
            Connecting to secure server...
          </div>
        )}
      </div>
    </>
  );
}
